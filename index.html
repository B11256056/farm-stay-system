async function getAnalysis(event) { if (event) event.preventDefault(); // 阻止頁面跳轉/重整，避免通訊中斷 const btn = document.getElementById('analyzeBtn'); const resultDiv = document.getElementById('result'); const textElement = document.getElementById('analysisText'); btn.disabled = true; btn.innerText = "分析中..."; resultDiv.classList.add('hidden'); try { // 重要：請將網址換成你 Vercel 實際的完整 Domain const API_ENDPOINT = 'https://farm-stay-system-h9dd.vercel.app/api/analyze'; const res = await fetch(API_ENDPOINT, { method: 'POST', headers: { 'Content-Type': 'application/json' } }); const data = await res.json(); resultDiv.classList.remove('hidden'); // 依照後端定義的 advice 欄位讀取 if (data.advice) { textElement.innerText = data.advice; } else { textElement.innerText = "收到空回應，請檢查後端日誌。"; } } catch (error) { resultDiv.classList.remove('hidden'); textElement.innerText = "連線失敗，請稍後再試。"; console.error("前端錯誤:", error); } finally { btn.disabled = false; btn.innerText = "產生 AI 經營分析"; } }
